// float double ve decimal ondalıklı değerleri barındırabilir
float   dolarToTL = 42.12f;
double  TLToDolar = 0.024;
decimal TLToEuro  = 0.021;

// float (veya single) değerler için sayının sonuna f koyulmalıdır, sonunda f olmayan ondalıklı sayılar otomatik olarak double olur
Console.WriteLine((0.0).GetType()); // get type değişkenin türünü söyler
// > System.Double
Console.WriteLine((0.0f).GetType());
// > System.Single
// sayının sonuna m koymak sayının decimal olduğunu ifade eder
// decimal ifade ederken sonuna m konması zorunludur
Console.WriteLine((0.0m).GetType());
// > System.Decimal

// ondalıklı sayı tipleri kesinliklerine göre ayrılır
// float 1 keskinliğindedir (adının single olmasının nedeni budur)
// double 2 keskinliğindedir 
// decimal 4 keskinliğindedir

// keskinliğin fazla olması ondalık değerinin daha net bir şekilde hesaplanmasını ifade eder
// float 7~ basamağın ardından güvenilir sonuçlar vermez
Console.WriteLine(1f / 3f);
// > 0.33333334

// double 15-16~ basamağa kadar hatasız kullanılabilir
Console.WriteLine(0.1 + 0.2);
// > 0.30000000000000004

// decimal 28-29 basamağa kadar kullanılabilir
Console.WriteLine(1m / 3m);
// > 0.3333333333333333333333333333
Console.WriteLine(0.1m + 0.2m);
// > 0.3

// float değerler ±1.5 x 10−45 ile ±3.4 x 1038 arasında olabilir
// double değerler ±5.0 × 10−324 ile ±1.7 × 10308 arasında olabilir
// decimal değerler ±1.0 x 10-28 to ±7.9228 x 1028 arasında olabilir

if ve else

// bir şarta göre kod çalıştırmak için kullanılır

// temel şartlar:
// a b'den küçük ise: a < b
// a b'den küçük veya eşit ise: a <= b
// a b'den büyük ise: a > b
// a b'den büyük veya b'ye eşit ise: a >= b
// a b'ye eşit ise: a == b
// a b'ye eşit değil ise: a != b

if (şart)
{
    // şart true (doğru) ise çalışacak olan kod
}

if (20 > 10)
{
    Console.WriteLine("20 10'dan büyüktür");
}
// > 20 10'dan büyüktür

// değişkenler içinde kullanılabilir
int x = 85;
int y = 25;
if (x > y)
{
    Console.WriteLine("x y'den büyüktür");
}
// > x y'den büyüktür

// metinlerin birbirine eşit olup olmadığını bilmek içinde kullanılabilir
string ad = "Hasan";
string soyAd = "El Hasan";


if (ad != soyAd)
{
    Console.WriteLine(ad + " ile " + soyAd + " birbirine eşit değildir");
}
// > Hasan ile El Hasan birbirine eşit değildir

// else, ifin içindeki şart true değil ise çalışır
if (20 > 80)
{
    Console.WriteLine("20 80'den büyüktür");
}
else
{
    Console.WriteLine("80 20'den büyüktür");
}
// > 80 20'den büyüktür

// else if, önceki if doğru değilse ve kendi şartı doğruysa çalışır
int gununSaati = 22;
if (gununSaati < 10)
{
    Console.WriteLine("Günaydın.");
}
else if (gununSaati < 20)
{ // günün saati 10dan büyük ve 20 den küçükse çalışır
    Console.WriteLine("İyi Günler.");
}
else
{
    Console.WriteLine("İyi Akşamlar.");
}
// > İyi Akşamlar.

for

for (ifade 1; ifade 2; ifade 3) 
{
  // ifade 2 true iken çalışıcak olan kod
}

// ifade 1: for çalışmadan önce ilk ve tek defa çalışacak olan kod
// ifade 2: kodun çalışmaya devam edip etmeyeceğini belirten şart
// ifade 3: kod bloğunun içindeki kod her çalıştıktan sonra çalışacak olan kod

for (
    int i = 0; // for çalışmadan önce i adında bir değilken oluştur ve değişkene 0 ata
    i < 5; // forun içindeki kod bloğunu i 5'den küçük olduğunca çalıştır
    i++ // kod bloğu her çalıştığında i'yi 1 arttır
    ) 
{
    Console.WriteLine(i);
}

// > 0
// > 1
// > 2
// > 3
// > 4
